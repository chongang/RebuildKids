       function setStatusMessage(message, type = 'success', element = null) {
           const messageElement = element || (document.querySelector('.tab.active .message')) ;
           if (messageElement) {
               messageElement.textContent = message;
               messageElement.className = `message ${type}`;
           } else {

           }, 3000);

       function handleCheckIn() {
           const studentName = document.getElementById('student').value;
           if (!studentName) {
               setStatusMessage("Please select a student", "error", checkMessage);
               return;
           }

           if (checkedInStudents.has(studentName)) {
               setStatusMessage("Student already checked in.", "error", checkMessage);
               return;
           }

           const timestamp = new Date().toISOString();
           const log = { student: studentName, checkInTime: timestamp, checkOutTime: null, lastUpdated: timestamp };

           let logs = JSON.parse(localStorage.getItem('logs') || '[]');
           logs.push(log);
           localStorage.setItem('logs', JSON.stringify(logs));
           checkedInStudents.add(studentName);

           updateLogTable();
           setStatusMessage(`Student ${studentName} checked in successfully!`, "success", checkMessage);
           document.getElementById('student').value = "";
       }

       function handleCheckOut(studentName) {
           const timestamp = new Date().toISOString();
           let logs = JSON.parse(localStorage.getItem('logs') || '[]');
           let studentLog = logs.find(log => log.student === studentName && log.checkOutTime === null);

           if (studentLog) {
               studentLog.checkOutTime = timestamp;
               studentLog.lastUpdated = timestamp;
               localStorage.setItem('logs', JSON.stringify(logs));
               checkedInStudents.delete(studentName);
               updateLogTable();
               setStatusMessage(`Student ${studentName} checked out successfully!`, "success", checkMessage);
           }
           else {
               setStatusMessage(`Student ${studentName} is not checked in!`, "error", checkMessage);
           }
       }

       function appendLogToTable(log) {
           const row = document.createElement('tr');
           row.innerHTML = `<td>${log.student}</td>
                            <td>${log.checkInTime ? new Date(log.checkInTime).toLocaleString() : '-'}</td>
                            <td>${log.checkOutTime ? new Date(log.checkOutTime).toLocaleString() : '-'}</td>
                            <td>${log.checkOutTime === null
                               ? `<button onclick="handleCheckOut('${log.student}')">⏹️ Check-Out</button>`
                               : `<button onclick="deleteLog('${log.student}')">🗑️ Delete</button>`}</td>`;
           row.onclick = () => {
               generateQRFromLog(log);
               updateSelectedRow(row);
           };
           logBody.appendChild(row);
       }

       function updateLogTable() {
           logBody.innerHTML = '';
           let logs = JSON.parse(localStorage.getItem('logs') || '[]');
           logs.sort((a, b) => {
               // Sort by check-in time, descending order (most recent first)
               const checkInComparison = b.checkInTime ? (a.checkInTime ? new Date(b.checkInTime) - new Date(a.checkInTime) : -1) : (a.checkInTime ? 1 : 0);
               if (checkInComparison !== 0) {
                   return checkInComparison;
               }
               // If check-in times are equal, sort by check-out time, descending order
               const checkOutComparison = b.checkOutTime ? (a.checkOutTime ? new Date(b.checkOutTime) - new Date(a.checkOutTime) : -1) : (a.checkOutTime ? 1 : 0);
               return checkOutComparison;
           });
           logs.forEach(log => appendLogToTable(log));
            if (selectedRow) {
               const selectedStudentName = selectedRow.cells[0].textContent;
               const newSelectedRow = Array.from(logBody.children).find(row => row.cells[0].textContent === selectedStudentName);
               if (newSelectedRow) {
                   updateSelectedRow(newSelectedRow);
               }
               else{
                   qrCanvas.classList.remove('qr-border-green', 'qr-border-red');
                   qrNameDisplay.textContent = '';
                   qrCheckInTimeDisplay.textContent = '';
                   qrCheckOutTimeDisplay.textContent = '';
                   qrStatusDisplay.textContent = '';
               }
           }
       }

       function generateQRFromLog(log) {
           const qrData = JSON.stringify(log);
           QRCode.toCanvas(qrCanvas, qrData, { width: 256 }, function (error) {
               if (error) {
                   console.error(error);
                   setStatusMessage("Error generating QR code.", "error", checkMessage);
               }
           });
           qrNameDisplay.textContent = log.student;
           qrCheckInTimeDisplay.textContent = log.checkInTime ? new Date(log.checkInTime).toLocaleString() : '-';
           qrCheckOutTimeDisplay.textContent = log.checkOutTime ? new Date(log.checkOutTime).toLocaleString() : '-';
           qrStatusDisplay.textContent = log.checkOutTime === null ? 'Checked-In' : 'Checked-Out';

           qrCanvas.classList.remove('qr-border-green', 'qr-border-red');
           if (log.checkOutTime === null) {
               qrCanvas.classList.add('qr-border-green');
           } else {
               qrCanvas.classList.add('qr-border-red');
           }
           qrInfoDiv.style.display = 'block';

       }

       function loadLogs() {
           let logs = JSON.parse(localStorage.getItem('logs') || '[]');
           logs.sort((a, b) => {
               // Sort by check-in time, descending order (most recent first)
               const checkInComparison = b.checkInTime ? (a.checkInTime ? new Date(b.checkInTime) - new Date(a.checkInTime) : -1) : (a.checkInTime ? 1 : 0);
               if (checkInComparison !== 0) {
                   return checkInComparison;
               }
               // If check-in times are equal, sort by check-out time, descending order
               const checkOutComparison = b.checkOutTime ? (a.checkOutTime ? new Date(b.checkOutTime) - new Date(a.checkOutTime) : -1) : (a.checkOutTime ? 1 : 0);
               return checkOutComparison;
           });
           logs.forEach(log => {
               if (log.checkOutTime === null) {
                   checkedInStudents.add(log.student);
               }
               appendLogToTable(log)
           });
       }

       }

       function addOrEditStudent() {
           const name = document.getElementById('newStudent').value.trim();
           if (!name) {
               setStatusMessage("Enter a valid student name.", "error", manageMessage);
               return;
           }
           let students = JSON.parse(localStorage.getItem('students') || '[]');
           if (!students.includes(name)) {
               students.push(name);
               localStorage.setItem('students', JSON.stringify(students));
               loadStudents(); // Reload students and update table
               document.getElementById('newStudent').value = '';
               setStatusMessage(`Student "${name}" added successfully!`, "success", manageMessage);
           } else {
               setStatusMessage(`Student "${name}" already exists.`, "error", manageMessage);
           }
       }

       function checkQueryParams() {
           const params = new URLSearchParams(window.location.search);
           const action = params.get('action');
           const student = params.get('student');
           if (action && student) {
               autoCheck(action, student);
           }
       }



       function populateStudentTable(students) {
           studentTableBody.innerHTML = '';
           students.forEach((name, index) => {
               const row = document.createElement('tr');
               const nameCell = document.createElement('td');
               nameCell.textContent = name;
               row.appendChild(nameCell);

               const actionsCell = document.createElement('td');
               actionsCell.classList.add('student-actions');

               const editButton = document.createElement('button');
               editButton.textContent = '✏️ Edit';
               editButton.classList.add('edit-button');
               editButton.addEventListener('click', () => {
                   console.log('Edit clicked for index:', index, 'name:', name);
                   editStudent(index, name);
               });
               actionsCell.appendChild(editButton);

               const deleteButton = document.createElement('button');
               deleteButton.textContent = '🗑️ Delete';
               deleteButton.classList.add('delete-button');
               deleteButton.addEventListener('click', () => {
                   console.log('Delete clicked for index:', index);
                   deleteStudent(index);
               });
               actionsCell.appendChild(deleteButton);

               row.appendChild(actionsCell);
               studentTableBody.appendChild(row);
           });
       }

       function editStudent(index, oldName) {
           console.log("editStudent function called with index:", index, "oldName:", oldName);
           const newName = prompt("Enter new name for " + oldName + ":", oldName);
           if (newName && newName.trim() !== oldName) {
               let students = JSON.parse(localStorage.getItem('students') || '[]');
               if (students.includes(newName.trim())) {
                   setStatusMessage("Student with this name already exists.", "error", manageMessage);
                   return;
               }
               students[index] = newName.trim();
               students.sort();
               localStorage.setItem('students', JSON.stringify(students));
               loadStudents();
               setStatusMessage(`Student name updated to ${newName}!`, "success", manageMessage);
           }
       }

       function deleteLog(studentName) {
           console.log("deleteLog function called with studentName:", studentName);
           if (confirm("Are you sure you want to delete this record?")) {
               let logs = JSON.parse(localStorage.getItem('logs') || '[]');
               const indexToDelete = logs.findIndex(log => log.student === studentName);
               if (indexToDelete !== -1) {
                   logs.splice(indexToDelete, 1);
                   localStorage.setItem('logs', JSON.stringify(logs));
                   updateLogTable();
                   setStatusMessage(`Log deleted successfully!`, "success", checkMessage);
               } else {
                   setStatusMessage(`Log not found for student ${studentName}!`, "error", checkMessage);
               }
           }
       }

       function deleteStudent(index) {
           console.log("deleteStudent function called with index:", index);
           if (confirm("Are you sure you want to delete this student?")) {
               let students = JSON.parse(localStorage.getItem('students') || '[]');
               students.splice(index, 1);
               localStorage.setItem('students', JSON.stringify(students));
               localStorage.setItem('logs', JSON.stringify([]));
               loadStudents();
               loadLogs();
               qrCanvas.classList.remove('qr-border-green', 'qr-border-red');
               qrNameDisplay.textContent = '';
               qrCheckInTimeDisplay.textContent = '';
               qrCheckOutTimeDisplay.textContent = '';
               qrStatusDisplay.textContent = '';
               setStatusMessage(`Student deleted successfully!`, "success", manageMessage);
           }
       }

       function updateSelectedRow(row) {
           if (selectedRow) {
               selectedRow.classList.remove('selected-row');
           }
           selectedRow = row;
           selectedRow.classList.add('selected-row');
       }

       window.onload = () => {
           loadLogs();
           loadStudents(); // Call loadStudents on window load
           checkQueryParams();
           qrInfoDiv.style.display = 'none';
       }
